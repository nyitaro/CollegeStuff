    private boolean isMouseIn(java.awt.Point mouse, javax.swing.JComponent area) {
        int x = mouse.x;
        int y = mouse.y;
        return ( x >= area.getX() && x <= area.getX() + area.getWidth() && 
                y >= area.getY() && y<= area.getY() + area.getHeight() );
    }
